from http.server import BaseHTTPRequestHandler, HTTPServer import json import os EGYENLEG_FAJL = "egyenlegek.json" if not os.path.exists(EGYENLEG_FAJL): with open(EGYENLEG_FAJL, "w") as f: json.dump({"milan": 5000, "teso": 5000}, f) class MilneyServer(BaseHTTPRequestHandler): def do_GET(self): # Ha a telefon kéri a HTML oldalt if self.path == "/": self.send_response(200) self.send_header("Content-type", "text/html; charset=utf-8") self.end_headers() # Beolvassuk a lenti HTML oldalt with open("index.html", "r", encoding="utf-8") as f: self.wfile.write(f.read().encode("utf-8")) def do_POST(self): # Ha a telefon elküldi a fizetési parancsot a közelség miatt if self.path == "/fizetes": self.send_response(200) self.send_header("Content-type", "application/json") self.end_headers() with open(EGYENLEG_FAJL, "r") as f: szamlak = json.load(f) # Levonunk 1000 Ft-ot Milántól if szamlak["milan"] >= 1000: szamlak["milan"] -= 1000 with open(EGYENLEG_FAJL, "w") as f: json.dump(szamlak, f) valasz = {"statusz": "siker", "uj_egyenleg": szamlak["milan"]} else: valasz = {"statusz": "hiba", "uzenet": "Nincs elég pénz!"} self.wfile.write(json.dumps(valasz).encode("utf-8")) print("=== Milney Helyi Szerver Elindult ===") server = HTTPServer(("0.0.0.0", 8080), MilneyServer) server.serve_forever()
